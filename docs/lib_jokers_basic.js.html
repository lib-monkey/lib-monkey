import Joker from '../common/joker';

/**
 * Boolean Joker
 * @class
 * @augments Joker
 * @inheritdoc
 * @property {number/bool} likelihood - likelihood of the char to be true.
 */
export class BooleanJoker extends Joker {

  static get name(){ return 'bool'; }

  get modifiers() {
    return ['likelihood'];
  }

  getValue() {
    return this.randomiser.bool(this.options);
  }

}

/**
 * Char Joker
 * @class

 * @property {bool} symbols - specify symbols for only symbols.

  static get name(){ return 'char'; }

}
